    <script>

        class SimpleChatbot {

            constructor() {

                this.messages = document.getElementById('messages');

                this.input = document.getElementById('messageInput');

                this.sendBtn = document.getElementById('sendBtn');

                this.uploadBtn = document.getElementById('uploadBtn');

                this.fileInput = document.getElementById('fileInput');

                this.uploadStatus = document.getElementById('uploadStatus');

                

                this.history = [];

                this.userId = this.generateUserId();

                this.isLoading = false;

                

                this.init();

            }

            

            init() {

                console.log('🤖 Chatbot starting...');

                

                this.sendBtn.addEventListener('click', () => this.handleSend());

                this.input.addEventListener('keypress', (e) => {

                    if (e.key === 'Enter' && !e.shiftKey) {

                        e.preventDefault();

                        this.handleSend();

                    }

                });

                this.uploadBtn.addEventListener('click', () => this.fileInput.click());

                this.fileInput.addEventListener('change', (e) => this.handleFileUpload(e));

                

                this.addMessage('👋 Hello! I\'m your UK Global Talent Visa assistant.\n\nI can help you with:\n• Eligibility requirements\n• Application process\n• Required documents\n• Timeline and costs\n• Endorsement bodies\n\nWhat would you like to know?', 'bot');

                

                this.testConnection();

                console.log('✅ Chatbot ready');

            }

            

            generateUserId() {

                return 'user_' + Math.random().toString(36).substr(2, 9);

            }

            

            async testConnection() {

                try {

                    const response = await fetch('/api/chat', {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify({ 

                            message: 'test', 

                            userId: this.userId,

                            history: []

                        })

                    });

                    

                    if (response.ok) {

                        console.log('✅ API connection successful');

                    } else {

                        console.warn('⚠️ API connection test failed');

                    }

                } catch (error) {

                    console.warn('⚠️ API not available:', error.message);

                }

            }

            

            async handleSend() {

                const message = this.input.value.trim();

                if (!message || this.isLoading) return;

                

                this.input.value = '';

                this.addMessage(message, 'user');

                

                this.isLoading = true;

                this.sendBtn.disabled = true;

                const loadingMsg = this.addMessage('Thinking...', 'bot', true);

                

                try {

                    const response = await fetch('/api/chat', {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify({

                            message: message,

                            userId: this.userId,

                            history: this.history.slice(-10)

                        })

                    });

                    

                    if (response.ok) {

                        const data = await response.json();

                        this.removeMessage(loadingMsg);

                        this.addMessage(data.response, 'bot');

                        

                        this.history.push(

                            { role: 'user', content: message },

                            { role: 'assistant', content: data.response }

                        );

                    } else {

                        throw new Error('API call failed');

                    }

                } catch (error) {

                    console.warn('API unavailable, using fallback responses');

                    this.removeMessage(loadingMsg);

                    

                    const fallbackResponse = this.getFallbackResponse(message);

                    this.addMessage(fallbackResponse, 'bot');

                }

                

                this.isLoading = false;

                this.sendBtn.disabled = false;

            }

            

            getFallbackResponse(message) {

                const msg = message.toLowerCase();

                

                if (msg.includes('eligib')) {

                    return '🎯 **UK Global Talent Visa Eligibility:**\n\n• Exceptional talent or promise in:\n  - Science, engineering, medicine\n  - Digital technology\n  - Arts and culture\n  - Research and academia\n\n• Endorsement from approved body\n• English proficiency\n• Financial requirements\n\nWhich field interests you most?';

                }

                

                if (msg.includes('cost') || msg.includes('fee')) {

                    return '💰 **UK Global Talent Visa Costs:**\n\n• Application fee: £623\n• Healthcare surcharge: £624/year\n• Endorsement fee: varies by field\n• Priority processing: +£500-£1000\n\nTotal typically £2000-£4000 depending on circumstances.';

                }

                

                if (msg.includes('timeline') || msg.includes('long')) {

                    return '⏰ **Processing Timeline:**\n\n• Endorsement: 8-12 weeks\n• Visa application: 3-8 weeks\n• Fast-track available: 5-10 days\n\nTotal process: 3-6 months typically\n\nStart endorsement application first!';

                }

                

                if (msg.includes('document')) {

                    return '📋 **Required Documents:**\n\n• Passport\n• Evidence of exceptional talent\n• CV and portfolio\n• Letters of recommendation\n• English test results\n• TB test (some countries)\n• Financial evidence\n\nSpecific requirements vary by endorsing body.';

                }

                

                return '🤖 I can help with UK Global Talent Visa questions about:\n\n• Eligibility requirements\n• Application process\n• Costs and timeline\n• Required documents\n• Endorsing bodies\n\nCould you be more specific about what you\'d like to know?';

            }

            

            async handleFileUpload(e) {

                const file = e.target.files[0];

                if (!file) return;

                

                if (file.type !== 'application/pdf') {

                    this.uploadStatus.textContent = 'Please upload a PDF file';

                    this.uploadStatus.style.color = '#dc3545';

                    return;

                }

                

                if (file.size > 5 * 1024 * 1024) {

                    this.uploadStatus.textContent = 'File too large (max 5MB)';

                    this.uploadStatus.style.color = '#dc3545';

                    return;

                }

                

                this.uploadStatus.textContent = 'Uploading...';

                this.uploadStatus.style.color = '#666';

                

                try {

                    const formData = new FormData();

                    formData.append('file', file);

                    formData.append('userId', this.userId);

                    

                    const response = await fetch('/api/upload', {

                        method: 'POST',

                        body: formData

                    });

                    

                    if (response.ok) {

                        this.uploadStatus.textContent = '✅ Uploaded successfully';

                        this.uploadStatus.style.color = '#28a745';

                        

                        this.addMessage('📄 I\'ve received your resume! I can now provide more personalized advice about your UK Global Talent Visa application. Feel free to ask specific questions!', 'bot');

                    } else {

                        throw new Error('Upload failed');

                    }

                } catch (error) {

                    this.uploadStatus.textContent = '❌ Upload failed';

                    this.uploadStatus.style.color = '#dc3545';

                    console.error('Upload error:', error);

                }

                

                this.fileInput.value = '';

            }

            

            addMessage(text, sender, isLoading = false) {

                const messageDiv = document.createElement('div');

                messageDiv.className = `message ${sender}-message`;

                if (isLoading) {

                    messageDiv.classList.add('loading', 'typing');

                    messageDiv.textContent = text;

                } else {

                    messageDiv.textContent = text;

                }

                

                this.messages.appendChild(messageDiv);

                this.messages.scrollTop = this.messages.scrollHeight;

                

                return messageDiv;

            }

            

            removeMessage(messageElement) {

                if (messageElement && messageElement.parentNode) {

                    messageElement.parentNode.removeChild(messageElement);

                }

            }

        }

        

        document.addEventListener('DOMContentLoaded', () => {

            new SimpleChatbot();

        });

    </script>
